Gradient Descent¶

Guess a solution $\vec x_0$

determine $$ \vec \triangle_0 = - \nabla f(\vec x_0) $$ update estimate $$ \vec x_1 = \vec x_0 + \eta \vec \triangle_0$$ repeat $$ \vec \triangle_t = - \nabla f(\vec x_t) $$ $$ \vec x_{t+1} = \vec x_t + \eta \vec \triangle_t$$

def grad_descent(x, y, lr=0.01):
    x_hist = np.array(x)

    for i in range(35):
        grad_x = grad_f(x, y)
        x = x - lr * grad_x
        x_hist = np.vstack((x_hist, x))
        
    return x_hist

Gradient Descent with Line Searches¶

Guess a solution $\vec x_0$

repeat $$ \vec \triangle_t = - \nabla f(\vec x_t) $$ $$ \eta_t = argmin_{\eta} \space f(\vec x_t + \eta \vec \triangle_t) $$ $$ \vec x_{t+1} = \vec x_t + \eta_t \vec \triangle_t$$

def grad_desc_line_search(x, y):
    
    x_hist = np.array(x)
    
    for i in range(100):
          
        change = -1 * grad_f(x, y)
        
        lr_list = np.linspace(0.,1,1000)
        min_value = f(x + lr_list[0] * change, y)
        min_lr = lr_list[0]

        for lr in lr_list:
            value = f(x + lr * change, y)
            if value < min_value:
                min_value = value
                min_lr = lr
        
        if min_lr!=0:
            x = x + min_lr * change
            x_hist = np.vstack((x_hist, x))
    
    return x_hist

Gradient Descent Using Conjugate Gradients¶

Guess a solution $\vec x_0$ $$ \vec \triangle_0 = - \nabla f(\vec x_0) $$ $$ \vec c_0 = \vec \triangle_0 $$ $$ \eta_0 = argmin_{\eta} \space f(\vec x_0 + \eta \vec c_0) $$ $$ \vec x_1 = \vec x_0 + \eta_0 \vec c_0$$

repeat $$ \vec \triangle_t = - \nabla f(\vec x_t) $$ $$ \vec c_t = \vec \triangle_t + \beta_t \vec c_{t-1} $$ $$ \eta_t = argmin_{\eta} \space f(\vec x_t + \eta \vec c_t) $$ $$ \vec x_{t+1} = \vec x_t + \eta_t \vec c_t$$ For beta: the Fletcher-Reeves formula: $$ \beta_t = max \bigg \{ 0, \frac{\vec \triangle_t^T \vec \triangle_t}{\vec \triangle_{t-1}^T \vec \triangle_{t-1}} \bigg \}$$ the Polak-Ribière formula:
$$ \beta_t = max \bigg \{ 0, \frac{\vec \triangle_t^T (\vec \triangle_t - \vec \triangle_{t-1})}{\vec \triangle_{t-1}^T \vec \triangle_{t-1}} \bigg \}$$

def f(x, y):
    return 2 * np.sum(np.log(LA.norm(x-y, axis=1)), axis=0)

def grad_f(x, y):
    return np.sum(2 * (x-y) / (LA.norm(x - y, axis=1).reshape(3,1))**2, axis=0)

def min_lr_search(x, y, change):
    
    lr_list = np.linspace(0.,1,1000)
    min_value = f(x + lr_list[0] * change, y)
    min_lr = lr_list[0]

    for lr in lr_list:
        value = f(x + lr * change, y)
        if value < min_value:
            min_value = value
            min_lr = lr
            
    return min_lr

"""
For beta:

Fletcher-Reeves formula
beta = np.max((0, (change[i].T @ change[i])/(change[i-1].T @ change[i-1])))

Polak-Ribière formula     
beta = np.max((0, (change[i].T @ (change[i] - change[i-1])) / (change[i-1].T @ change[i-1]) ))
"""
def grad_desc_conj_grad(x, y):
    
    c = []
    change = []
    x_hist = np.array(x)
    
    change.append(-1 * grad_f(x, y)) 
    c.append(change[0])
    min_lr = min_lr_search(x, y, c[0])

    x = x + min_lr * c[0]
    x_hist = np.vstack((x_hist, x))
    
    
    for i in range(1,30):
        
        change.append(-1 * grad_f(x, y))
        
        beta = np.max((0, (change[i].T @ (change[i] - change[i-1])) / (change[i-1].T @ change[i-1]) ))  #PR
        #beta = np.max((0, (change[i].T @ change[i])/(change[i-1].T @ change[i-1])))  #FR
          
        c.append(change[i] + beta * c[i-1])

        min_lr = min_lr_search(x, y, c[i])
        
        if min_lr != 0: 
            x = x + min_lr * c[i]
            x_hist = np.vstack((x_hist, x))
    
    return x_hist

y = np.array([[1., 0.], [-1., 1.], [-1.5, -1.]])

def f_th(x): 
    y = th.tensor([[1., 0.], [-1., 1.], [-1.5, -1.]])
    return 2 * th.sum(th.log(th.norm(x-y, dim=1)), dim=0)

def grad_f(x):
    y = np.array([[1., 0.], [-1., 1.], [-1.5, -1.]])
    return np.sum(2 * (x-y) / (LA.norm(x - y, axis=1).reshape(3,1))**2, axis=0)

def newtons_method(x, lr):

    x_hist = np.array(x)

    for i in range(5):
        change = -1 * grad_f(x)
        hessian = th.autograd.functional.hessian(f_th, th.tensor(x))

        x = x - lr * LA.inv(hessian.numpy()) @ change
        x_hist = np.vstack((x_hist, x))

    return x_hist
